Eleventy Baseline is a plugin you load in your `.eleventy.config.js`. It brings its own defaults but tries to keep room for your own config.

Right now, some wiring is opinionated/hard-wired (entrypoints, head tag, assets) while the project is still evolving; this is how it’s currently used day to day.

## Baseline plugin
In your root config:
```js
import baseline from "./_baseline/eleventy.config.js";
import { config as _config } from "./_baseline/eleventy.config.js";

export default function (eleventyConfig) {
  eleventyConfig.addPlugin(baseline({
    verbose: true,               // show extra logs
    Bob: "My number one guy.",   // example custom flag surfaced in _baseline
  }));
}

export const config = _config; // reuse Baseline’s dir/template defaults
```

Adding the Baseline plugin essentially imports a fairly classic eleventy config file. Reusing the plugins config object is for convenience.

This is what the current Baseline configuration looks like:
```js
/* Site config */
import "dotenv/config";

import debug from "./debug.js";
import filters from "./filters.js";
import modules from "./modules.js";
import shortcodes from "./shortcodes.js";

/**
 * Eleventy Baseline Plugin (factory-style)
 * @param {object} options - Custom options for the plugin.
 * @returns {(eleventyConfig: UserConfig) => void}
 */
export default function baseline(options = {}) {
	/** @param {import("@11ty/eleventy").UserConfig} eleventyConfig */
	return async function(eleventyConfig) {
		try {
			// Emit a warning message if the application is not using Eleventy 3.0 or newer (including prereleases).
			eleventyConfig.versionCheck(">=3.0");
		} catch(e) {
			console.log( `[eleventy-baseline] WARN Eleventy plugin compatibility: ${e.message}` );
		}

		const userOptions = { 
			verbose: options.verbose ?? false,
			...options 
		};

		eleventyConfig.addGlobalData("_baseline", userOptions);

		// Filters.
		eleventyConfig.addFilter("markdownify", filters.markdownFilter);
		eleventyConfig.addFilter("relatedPosts", filters.relatedPostsFilter);
		eleventyConfig.addFilter("inlinePostCSS", filters.inlinePostCSS);
		eleventyConfig.addFilter("inlineESbuild", filters.inlineESbuild);
		eleventyConfig.addFilter("isString", filters.isStringFilter);

		// Passthrough copy.
		eleventyConfig.addPassthroughCopy({ "./src/static": "/" });

		// Modules.
		eleventyConfig.addPlugin(modules.EleventyHtmlBasePlugin, { baseHref: process.env.URL || "/" });
		eleventyConfig.addPlugin(modules.syntaxHighlight);
		eleventyConfig.addPlugin(modules.assetsCore);
		eleventyConfig.addPlugin(modules.assetsPostCSS);
		eleventyConfig.addPlugin(modules.assetsESBuild);
		eleventyConfig.addPlugin(modules.navigatorCore);
		eleventyConfig.addPlugin(modules.navigatorContext);
		eleventyConfig.addPlugin(modules.headCore);

		// Shortcodes.
		eleventyConfig.addShortcode("image", shortcodes.imageShortcode);

		// Watch target.
		eleventyConfig.addWatchTarget('./src/assets/**/*.{css,js,svg,png,jpeg}');

		// Debug filters.
		eleventyConfig.addFilter("inspect", debug.inspect);
		eleventyConfig.addFilter("json", debug.json);
		eleventyConfig.addFilter("keys", debug.keys);
	};
}

export const config = {
	dir: {
		input: "src",
		output: "dist",
		data: "_data",
		includes: "_includes",
		assets: "assets"
	},
	htmlTemplateEngine: "njk",
	markdownTemplateEngine: "njk",
	templateFormats: ["html", "njk", "md"]
};
```

### Directories
Baseline exports Eleventy dir defaults and creates a virtual directory for assets:
```
input: src
output: dist
data: _data
includes: _includes
assets: assets
```

### Head/meta defaults
`<baseline-head>` is injected via PostHTML. Defaults merge site/page data with `_data/head.js` (link/script) and set title/description/robots + OG/Twitter shells. You can extend via page/front matter `head:` or by editing `_data/head.js`.

### Site data
`_data/site.js` holds title/tagline/baseURL/defaultLanguage/noindex. Override per page in front matter as needed.

### Customizing further
You can still add your own filters, shortcodes, collections, and plugins in your root config. Baseline is meant to be additive, not exclusive.

I recommend creating an `_extend` directory, put all your customisations there and then import them like you are used to in your `.eleventy.config.js` file.
This keeps your overrides separate from Baseline’s files and makes future updates easier.
